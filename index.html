function launch() { // We use the proxy so the network doesn't know we are hitting GitHub const target = "https://cdn.jsdelivr.net/gh/gn-math/gn-math-DONTDMCA@main/singlefile.html"; const proxy = "https://corsproxy.io/?" + encodeURIComponent(target); fetch(proxy) .then(response => response.text()) .then(html => { // Clear the page document.open(); document.write(html); document.close(); // This 'document.write' method actually handles script activation // better than innerHTML in many browsers! }) .catch(error => console.error('Bridge failed:', error)); }
